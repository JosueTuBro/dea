    // I) PRIMOS (7 métodos)
    // 1) primo(n) – versión O(√n)
    public static boolean primo(int n) {
        if (n <= 1) return false;
        return primoAux(n, 2);
    }
    private static boolean primoAux(int n, int d) {
        if ((long)d * d > n) return true;
        if (n % d == 0) return false;
        return primoAux(n, d + 1);
    }

    // 2) mostrarPrimos(a, b)
    public static void mostrarPrimos(int a, int b) {
        if (a > b) return;
        if (primo(a)) System.out.print(a + " ");
        mostrarPrimos(a + 1, b);
    }

    // 3) cantidadPrimos(a, b)
    public static int cantidadPrimos(int a, int b) {
        if (a > b) return 0;
        return (primo(a) ? 1 : 0) + cantidadPrimos(a + 1, b);
    }

    // 4) siguientePrimo(n)
    public static int siguientePrimo(int n) {
        if (primo(n)) return n;
        return siguientePrimo(n + 1);
    }

    // 5) anteriorPrimo(n)
    public static int anteriorPrimo(int n) {
        if (n < 2) return -1;
        if (primo(n)) return n;
        return anteriorPrimo(n - 1);
    }

    // 6) iesimoPrimo(i)
    public static int iesimoPrimo(int i) {
        return iesimoPrimoAux(i, 2, 0);
    }
    private static int iesimoPrimoAux(int i, int cand, int cont) {
        if (primo(cand)) {
            if (cont + 1 == i) return cand;
            return iesimoPrimoAux(i, cand + 1, cont + 1);
        }
        return iesimoPrimoAux(i, cand + 1, cont);
    }

    // 7) mostrarFactoresPrimos(n)
    public static void mostrarFactoresPrimos(int n) {
        mostrarFactoresPrimosAux(Math.abs(n), 2);
    }
    private static void mostrarFactoresPrimosAux(int n, int f) {
        if (n <= 1) return;
        if (n % f == 0) {
            System.out.print(f + " ");
            mostrarFactoresPrimosAux(n / f, f);
        } else {
            mostrarFactoresPrimosAux(n, f + 1);
        }
    }


    // II) CADENAS
    // Aux: seEncuentra(s, ch)
    public static boolean seEncuentra(String s, char ch) {
        if (s.isEmpty()) return false;
        if (s.charAt(0) == ch) return true;
        return seEncuentra(s.substring(1), ch);
    }

    // 1) frecuencia(s, ch)
    public static int frecuencia(String s, char ch) {
        if (s.isEmpty()) return 0;
        return (s.charAt(0) == ch ? 1 : 0) + frecuencia(s.substring(1), ch);
    }

    // 2) estanTodos(s1, s2)
    public static boolean estanTodos(String s1, String s2) {
        if (s1.isEmpty()) return true;
        if (!seEncuentra(s2, s1.charAt(0))) return false;
        return estanTodos(s1.substring(1), s2);
    }

    // 3) mismosChar(s1, s2)
    public static boolean mismosChar(String s1, String s2) {
        return estanTodos(s1, s2) && estanTodos(s2, s1);
    }

    // 4) palindrome(s)
    public static boolean palindrome(String s) {
        if (s.length() <= 1) return true;
        if (s.charAt(0) != s.charAt(s.length() - 1)) return false;
        return palindrome(s.substring(1, s.length() - 1));
    }

    // 5) existeVocal(s)
    public static boolean existeVocal(String s) {
        if (s.isEmpty()) return false;
        char c = Character.toLowerCase(s.charAt(0));
        if (c=='a'||c=='e'||c=='i'||c=='o'||c=='u') return true;
        return existeVocal(s.substring(1));
    }

    // 6) frecuenciaDigitos(s)
    public static int frecuenciaDigitos(String s) {
        if (s.isEmpty()) return 0;
        return (Character.isDigit(s.charAt(0)) ? 1 : 0) + frecuenciaDigitos(s.substring(1));
    }

    // 7) ordenado(s) – asc o desc (no estricto)
    public static boolean ordenado(String s) {
        return ordenAsc(s) || ordenDesc(s);
    }
    private static boolean ordenAsc(String s) {
        if (s.length() <= 1) return true;
        if (s.charAt(0) > s.charAt(1)) return false;
        return ordenAsc(s.substring(1));
    }
    private static boolean ordenDesc(String s) {
        if (s.length() <= 1) return true;
        if (s.charAt(0) < s.charAt(1)) return false;
        return ordenDesc(s.substring(1));
    }

    public static void tablaCoprimos(int n) { 
        tablaCoprimosAux(n, 1, 1); 
    }
    private static void tablaCoprimosAux(int n, int i, int j) {
        if (i > n) return;
        if (j > n) { tablaCoprimosAux(n, i + 1, 1); return; }
        if (mcd(i, j) == 1) {
            System.out.println(i + " x " + j + " = " + (i*j));
        }
        tablaCoprimosAux(n, i, j + 1);
    }
    private static int mcd(int a, int b) { 
        return (b == 0) ? Math.abs(a) : mcd(b, a % b);
    }
